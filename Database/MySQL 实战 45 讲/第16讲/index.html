    16 вҖңorder byвҖқжҳҜжҖҺд№Ҳе·ҘдҪңзҡ„пјҹ¶

    еңЁдҪ ејҖеҸ‘еә”з”Ёзҡ„ж—¶еҖҷпјҢдёҖе®ҡдјҡз»Ҹеёёзў°еҲ°йңҖиҰҒж №жҚ®жҢҮе®ҡзҡ„еӯ—ж®өжҺ’еәҸжқҘжҳҫзӨәз»“жһңзҡ„йңҖжұӮгҖӮиҝҳжҳҜд»ҘжҲ‘д»¬еүҚйқўдёҫдҫӢз”ЁиҝҮзҡ„еёӮж°‘иЎЁдёәдҫӢпјҢеҒҮи®ҫдҪ иҰҒжҹҘиҜўеҹҺеёӮжҳҜвҖңжқӯе·һвҖқзҡ„жүҖжңүдәәеҗҚеӯ—пјҢе№¶дё”жҢүз…§е§“еҗҚжҺ’еәҸиҝ”еӣһеүҚ 1000 дёӘдәәзҡ„е§“еҗҚгҖҒе№ҙйҫ„гҖӮ

    еҒҮи®ҫиҝҷдёӘиЎЁзҡ„йғЁеҲҶе®ҡд№үжҳҜиҝҷж ·зҡ„пјҡ

    CREATE TABLE `t` (
  `id` int(11) NOT NULL,
  `city` varchar(16) NOT NULL,
  `name` varchar(16) NOT NULL,

) ENGINE=InnoDB;

    еүҚйқўжҲ‘д»¬д»Ӣз»ҚиҝҮзҙўеј•пјҢжүҖд»ҘдҪ зҺ°еңЁе°ұеҫҲжё…жҘҡдәҶпјҢдёәйҒҝе…Қе…ЁиЎЁжү«жҸҸпјҢжҲ‘д»¬йңҖиҰҒеңЁ city еӯ—ж®өеҠ дёҠзҙўеј•гҖӮ

    еңЁ city еӯ—ж®өдёҠеҲӣе»әзҙўеј•д№ӢеҗҺпјҢжҲ‘д»¬з”Ё explain е‘Ҫд»ӨжқҘзңӢзңӢиҝҷдёӘиҜӯеҸҘзҡ„жү§иЎҢжғ…еҶөгҖӮ

    дёәдәҶиҜҙжҳҺиҝҷдёӘ SQL жҹҘиҜўиҜӯеҸҘзҡ„жү§иЎҢиҝҮзЁӢпјҢжҲ‘д»¬е…ҲжқҘзңӢдёҖдёӢ city иҝҷдёӘзҙўеј•зҡ„зӨәж„ҸеӣҫгҖӮ

    д»ҺеӣҫдёӯеҸҜд»ҘзңӢеҲ°пјҢж»Ўи¶і city='жқӯе·һвҖҷжқЎд»¶зҡ„иЎҢпјҢжҳҜд»Һ ID_X еҲ° ID_(X+N) зҡ„иҝҷдәӣи®°еҪ•гҖӮ

    йҖҡеёёжғ…еҶөдёӢпјҢиҝҷдёӘиҜӯеҸҘжү§иЎҢжөҒзЁӢеҰӮдёӢжүҖзӨә пјҡ
